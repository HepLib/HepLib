52 bool is_complete(const ex & props, const ex & ips, const ex & eps) {
53 auto sp2x = _sp2x_(ips, eps);
54 auto ps = subs(props, sp2x);
55 int n = ps.nops();
56 matrix mat(n,n);
57 for(int r=0; r<n; r++) {
58 auto xi = x(r);
59 for(int c=0; c<n; c++) {
60 mat(r,c) = ps.op(c).coeff(xi);
