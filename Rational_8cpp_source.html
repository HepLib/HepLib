53 numeric ChineseRemainder(std::vector<numeric> a, std::vector<numeric> n) {
54 numeric prod = 1;
55 for (int i = 0; i < n.size(); i++) prod *= n[i];
56
57 numeric sm = 0;

59 numeric p = iquo(prod, n[i]);
